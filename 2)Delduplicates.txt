public class Delduplicates
{
   public static void main (String[] args)
    {
     int a[]=new int[]{1,2,3,1,2,3},count=0;
     for(int x=0;x<a.length;x++)
       {
        count=0;
        for(int y=x+1;y<a.length;y++)
          {
            if(a[x]==a[y])
             {
              count++;
             }
          }
         if(count==0)
         System.out.print(a[x]+" ");
}           
 }
}
